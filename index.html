function initMap() { navigator.geolocation.getCurrentPosition(position => { userLocation = [position.coords.longitude, position.coords.latitude]; map = tt.map({ key: apiKey, container: 'map', center: userLocation, zoom: 14 }); map.addControl(new tt.NavigationControl()); new tt.Marker().setLngLat(userLocation).addTo(map); }); } function searchNearby(query) { if (!userLocation) return alert("Location not available yet."); const [lon, lat] = userLocation; fetch(`https://api.tomtom.com/search/2/poiSearch/${query}.json?key=${apiKey}&lat=${lat}&lon=${lon}&radius=10000`) .then(res => res.json()) .then(data => { data.results.forEach(poi => { new tt.Marker() .setLngLat([poi.position.lon, poi.position.lat]) .setPopup(new tt.Popup().setText(poi.poi.name)) .addTo(map); }); }); } initMap();
